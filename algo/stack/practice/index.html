    !popOrder ||
    pushOrder.length === 0 ||
    popOrder.length === 0
  ) {
    return false;
  }

  let index = 0;
  for (let i = 0, length = pushOrder.length; i < length; i++) {
    stack.push(pushOrder[i]);
    while (stack.length && stack[stack.length - 1] === pushOrder[index]);
