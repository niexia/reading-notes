class LinkedListQueue {
  constructor() {
    this.head = null;
    this.tail = null;
  }
  enqueue(element) {
    if (this.head === null) {
      this.head = new Node(element);
      this.tail = this.head;
    } else {
      this.tail.next = new Node(element);
      this.tail = this.tail.next;
    }
  }
  dequeue() {
    if (this.head === null) {
      return -1;
    } else {
      const value = this.head.element;
      this.head = this.head.next;
      return value;
    }
  }
}
